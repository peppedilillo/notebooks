from astropy.table import Table

def fits2dataframe(filepath):
    t = Table.read(filepath)
    d = np.array(t).byteswap().newbyteorder()
    df = t.to_pandas()
    df["EVTYPE"] = df["EVTYPE"].map({b"X": "X", b"S": "S"})
    df["QUADID"] = df["QUADID"].map({b"A": "A", b"B": "B", b"C": "C", b"D": "D"})
    return df

def normal(x, amp, sigma, x0): 
    o = amp * np.exp(-((x - x0) ** 2) / (2 * sigma**2)) / (sigma * np.sqrt(2 * np.pi))
    return o

Geometric area¶

We define the geometric area as: $$ A = ( N - 2n ) a $$

where $N = 120$ is the total number of detector channels, $n$ is the number of scintillators coupled to two off channels, and $a = 6.05\times7.44$ mm$^2$ is the area of an individual SDD cell.

The geometric area fraction is:

$$ F = \frac{A}{N a} = 1 - \frac{2 n}{N}$$

def geometric_area(payload):
    quad_scint_num = 120/4/2
    sdd_cal = pd.read_excel(datafolder[payload].joinpath("report_cal.xlsx"), index_col=0, sheet_name=None)
        
    n_ = 0
    for quad in "ABCD":
        channels = sdd_cal[quad].index.values
        unique_scint_ids = len(set([detectors[payload].scintid(quad, ch) for ch in channels]))
        quadn_ = quad_scint_num - unique_scint_ids;
        n_ += quadn_
    return (N - 2*n_) * a / 100

Other approaches¶

Let's consider two alternative definitions of geometric area.

  1. $A_x = (N - n') a$ where $n'$ is the number of off channels.
  2. $A_s = n''a$ where $n''$ is the number of channels successfully s-calibrated.
In [31]:
def geometric_area_x(payload):
    quad_sdd_num = 120/4
    n_ = 0
    sdd_cal = pd.read_excel(datafolder[payload].joinpath("report_cal.xlsx"), index_col=0, sheet_name=None)
    for quad in "ABCD":
        channels = sdd_cal[quad].index.values
        quadn_ = quad_sdd_num - len(channels);
        n_ += quadn_
    return (N - n_) * a / 100
